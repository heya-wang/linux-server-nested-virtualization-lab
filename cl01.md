> Role: Router / Jump Host

## Configuration as SSH jump host

```bash
# Generate SSH key pair (RSA 4096)
ssh-keygen -t rsa -b 4096

# Copy public key to server
ssh-copy-id -i ~/.ssh/id_rsa.pub student@192.168.10.11
ssh-copy-id -i ~/.ssh/id_rsa.pub student@192.168.10.13
ssh-copy-id -i ~/.ssh/id_rsa.pub student@192.168.10.14

# Test SSH login
ssh student@192.168.10.11
ssh student@192.168.10.13
ssh student@192.168.10.14

# Keys are stored in:
~/.ssh/id_rsa      # Private key
~/.ssh/id_rsa.pub  # Public key

# Create and edit SSH config file
nano ~/.ssh/config
ssh-config

# Default configuration for all hosts
Host *
    User student
    IdentityFile ~/.ssh/id_rsa
    Port 22

Host srv1
    HostName 192.168.10.11
Host srv2
    HostName 192.168.10.13
Host srv3
    HostName 192.168.10.14

# Test connections
ssh srv1
ssh srv2
ssh srv3

# Install and enable SSH server (as ssh jump host)
sudo apt update
sudo apt install openssh-server -y
sudo systemctl enable ssh
sudo systemctl start ssh
sudo systemctl status ssh

#On Windows Host - Connect to CL01 jump host
ssh student@10.100.10.1

```

## Configuration as LDAP Client 

```bash
# Install
sudo apt update
sudo apt install libnss-ldapd libpam-ldapd ldap-utils -y

# During install:
# URI: ldap://192.168.10.11
# Base: dc=gfn,dc=internal
# Services: passwd, group, shadow

# NSS config
sudo nano /etc/nsswitch.conf
# Ensure:
# passwd: files systemd ldap
# group:  files systemd ldap
# shadow: files ldap

# Auto create home
echo "session optional pam_mkhomedir.so skel=/etc/skel umask=0022" \
| sudo tee -a /etc/pam.d/common-session

# Restart
sudo systemctl restart nslcd
sudo systemctl restart nscd

# Test
getent passwd hwang
id hwang
su - hwang
```
